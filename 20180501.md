#include <iostream>
using namespace std;

class Sphere
{
public:
  // 球的半徑
  double radius;

  // 若無半徑 值=1
  Sphere()
  
    radius = 1;
  }

  // 設定一個球的半徑 
  Sphere(double newRadius)
  {
    radius = newRadius;
  }

  // 球體積的計算 
  double getArea()
  {
    return 1.333*radius*radius*radius*3.14159;
  }
};  // 必須放置一個分號 

int main()
{
  Sphere Sphere1(1.0);
  Sphere Sphere2(25);
  Sphere Sphere3(125);

  cout << "球的半徑 "   << Sphere1.radius << "的體積 " << Sphere1.getArea() << endl;
  cout << "球的半徑 "   << Sphere2.radius << "的體積 " << Sphere2.getArea() << endl;
  cout << "球的半徑 "  << Sphere3.radius << "的體積 " << Sphere3.getArea() << endl;

  //球半徑的數值
  Sphere2.radius = 100;
  cout << "球的半徑 "   << Sphere2.radius << "的體積" << Sphere2.getArea() << endl;

  return 0;
}
